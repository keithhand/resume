% {Category}{Skill}
\newcommand{\skill}[2]{
  \setsection{1}
  \section{#1}
  % #2
  \begin{enumerate*}[label=, before=\small, itemjoin={,}, style=sameline]
    #2
  \end{enumerate*}
}

% [End date]{Start date}{Title}{Company}{Location}{Description}
\newcommand{\experience}[6][Present]{
  \setsection{2}
  \setstaticbullets{}
  \section{#3}
    \formatdate{#2 -- #1}
  \subsection{{\normalsize#4} - {\footnotesize\itshape#5}}
  \begin{description}[style=standard, before=\footnotesize, itemsep={.5ex}, leftmargin={2em}, itemindent={-.7em}, labelsep={0em}, rightmargin=\leftmargin]
    #6
  \end{description}
}
\newcommand{\setstaticbullets}[1]{
  \renewcommand\descriptionlabel[1]{$\bullet$ {#1}}
}

% [Minor Degree]{Major Degree}{Graduation date}{University}
\newcommand{\education}[4][]{
  \setsection{2}
  \section{
    #2
    \ifx#1\else{\mdseries\footnotesize\itshape - Minor in #1}\fi
  }
    \formatdate{#3}
  \subsection{
    #4
  }
  \hspace{0em}
}
